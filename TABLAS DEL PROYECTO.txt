create table CUENTA
(
   IDCUENTA             int not null,
   SALDO                numeric(10,2),
   primary key (IDCUENTA)
);

-------------------------------------------------

create table MOVIMIENTO
(
   IDMOVIMIENTO         int not null,
   IDCUENTA             int,
   TIPOMOVIMIENTO       varchar(2),
   VALOR                numeric(10,2),
   primary key (IDMOVIMIENTO)
);

--------------------------------------------------

create table TRANSFERENCIA
(
   IDCUENTA             int not null,
   IDCUENTAORIGEN       int,
   IDCUENTADESTINO      int,
   IMPORTE              numeric(10,2)
);

alter table MOVIMIENTO add constraint FK_RELATIONSHIP_2 foreign key (IDCUENTA)
      references CUENTA (IDCUENTA) on delete restrict on update restrict;

alter table TRANSFERENCIA add constraint FK_RELATIONSHIP_3 foreign key (IDCUENTA)
      references CUENTA (IDCUENTA) on delete restrict on update restrict;
